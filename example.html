How To Use

  <title>Example</title>

  <script type="text/javascript" src="nerine.js"></script>
  <link rel="stylesheet" type="text/css" href="nerine.css">
</head><body>
<article>
  <section>
    write some page
  </section>
</article>
</body></html>
